% -----------------------------------------------------------------------------
%  目录 / Table of Contents
%  ─────────────────────────
%  由 \section / \subsection 命令自动生成，需编译两次页码才正确。
%  Auto-generated from \section/\subsection. Requires two compilations to
%  resolve page numbers correctly.
%  目录页不计入正文 "Page X of Y" 的页码范围。
%  The ToC page is excluded from the "Page X of Y" body count.
% -----------------------------------------------------------------------------
\pagestyle{empty}           % 目录页无页眉页脚 / No header/footer on ToC page
\tableofcontents
\thispagestyle{empty}
\clearpage

% 恢复 XMUM 页眉页脚，正文页码从第 1 页开始。
% Restore XMUM header/footer; body page count starts from 1.
% "Page X of Y" 中的 Y 以 \label{LastMainPage} 为终点，附录不计入。
% Y in "Page X of Y" ends at \label{LastMainPage} — appendices excluded.
\pagestyle{fancy}
\fancyfoot[C]{Page \thepage\ of \pageref{LastMainPage}}
\setcounter{page}{1}

% -----------------------------------------------------------------------------
%  正文内容 / Main Body Content
%  ─────────────────────────────
%  每个 section 一个 .tex 文件，用 \input 引入（强烈推荐，见 overview.tex）。
%  One .tex file per section, included via \input (strongly recommended,
%  see overview.tex for the rationale).
% -----------------------------------------------------------------------------
\input{sections/overview.tex}
\input{sections/sample_section.tex}
\input{sections/best_practices.tex}

% -----------------------------------------------------------------------------
%  参考文献 / Bibliography
%  ────────────────────────
%  在 ref.bib 中用 BibTeX 格式添加文献，模板自动应用 APA 格式。
%  Add references to ref.bib in BibTeX format; APA style is applied automatically.
%  "References" 标题会自动加入目录。
%  The "References" heading is added to the ToC automatically.
% -----------------------------------------------------------------------------
\bibliography{ref}

% -----------------------------------------------------------------------------
%  正文结束标记 / End-of-Main-Body Marker
%  ─────────────────────────────────────
%  必须放在 \bibliography 之后、\clearpage 之前，用于 "Page X of Y" 计算。
%  Must remain AFTER \bibliography and BEFORE \clearpage — used by the footer.
% -----------------------------------------------------------------------------
\label{LastMainPage}
\clearpage

% -----------------------------------------------------------------------------
%  附录 / Appendix
%  ────────────────
%  附录标题自动编号为 A、B、C……
%  Appendix sections are labelled A, B, C… automatically.
%
%  页码重置为罗马数字（i, ii, iii…），与正文独立计数，符合英文论文规范。
%  Page numbering restarts in Roman numerals (i, ii, iii…), independent of
%  the main body — following standard academic convention.
%  页脚只显示数字本身，不含 "of N"。
%  Footer shows only the numeral, no "of N".
% -----------------------------------------------------------------------------
\pagenumbering{roman}
\fancyfoot[C]{\thepage}

\input{sections/appendix.tex}

\clearpage
